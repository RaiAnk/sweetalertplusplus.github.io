Email Input

With Validation
JavaScript 
const email = await Swal.prompt({
  title: 'Subscribe to newsletter',
  text: 'Get updates on new features',
  inputType: 'email',
  placeholder: 'you@example.com',
  style: 'aurora',
  validate: (value) => {
    if (!value.includes('@')) {
      return 'Please enter a valid email';
    }
  }
});
