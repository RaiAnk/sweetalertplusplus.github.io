Installation

Choose your preferred installation method

NPM / Yarn / PNPM 

# npm
npm install sweetalert-plus-plus

# yarn
yarn add sweetalert-plus-plus

# pnpm
pnpm add sweetalert-plus-plus

Then import in your JavaScript:

import { modal, alert, confirm, toast } from 'sweetalert-plus-plus';
import 'sweetalert-plus-plus/css';

CDN (jsDelivr / unpkg)

Add these to your HTML <head>:

<!-- CSS -->
<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/sweetalert-plus-plus@1/dist/sweetalert-plus-plus.css">

<!-- JavaScript -->
<script src="https://cdn.jsdelivr.net/npm/sweetalert-plus-plus@1/dist/sweetalert-plus-plus.min.js"></script>

Or use unpkg:

<link rel="stylesheet" href="https://unpkg.com/sweetalert-plus-plus@1/dist/sweetalert-plus-plus.css">
<script src="https://unpkg.com/sweetalert-plus-plus@1/dist/sweetalert-plus-plus.min.js"></script>

Quick Start (HTML) 

<!DOCTYPE html>
<html>
<head>
  <link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/sweetalert-plus-plus@1/dist/sweetalert-plus-plus.css">
</head>
<body>
  <button onclick="SweetAlertPlusPlus.alert('Hello World!')">Show Alert</button>
  <button onclick="SweetAlertPlusPlus.confirm('Are you sure?').then(r => console.log(r))">Confirm</button>
  <button onclick="SweetAlertPlusPlus.toast('Notification!')">Toast</button>

  <script src="https://cdn.jsdelivr.net/npm/sweetalert-plus-plus@1/dist/sweetalert-plus-plus.min.js"></script>
</body>
</html>

Prompt Examples 
// Text Input Prompt
const nameResult = await Swal.modal({
  title: 'What is your name?',
  input: 'text',
  inputPlaceholder: 'Enter your name...',
  buttons: {
    cancel: 'Cancel',

// Email Input
const emailResult = await Swal.modal({
  title: 'Enter your email',
  input: 'email',
  inputPlaceholder: 'your@email.com',
  inputValidator: (value) => {
    if (!value.includes('@')) {
      return 'Please enter a valid email address';
    }
  }

const passResult = await Swal.modal({
  title: 'Enter Password',
  input: 'password',
  inputPlaceholder: 'Your password...',
  inputAttributes: {
    minlength: 8,
    autocapitalize: 'off',
    autocorrect: 'off'
  }
});

// With Validation
const validatedResult = await Swal.modal({
  title: 'Enter a URL',
  input: 'url',
  inputPlaceholder: 'https://example.com',
  inputValidator: (value) => {
    try {
      new URL(value);
    } catch {
      return 'Please enter a valid URL';
    }
  },
  showCancelButton: true
});

});

// Toast Positions
Swal.toast({
  icon: 'success',
  title: 'Top Right',
  position: 'top-end'  // Options: top-start, top-center, top-end,
                       // bottom-start, bottom-center, bottom-end
});

// Toast with Actions
Swal.toast({
  icon: 'info',
  title: 'New Message',
  text: 'You have a new message.',
  buttons: {
    view: {
      text: 'View',
      onClick: () => console.log('View clicked')
    },
    dismiss: 'Dismiss'
  }
});

// Persistent Toast (no auto-dismiss)
Swal.toast({
  icon: 'warning',
  title: 'Attention Required',
  text: 'Please review this item.',
  duration: 0,  // Won't auto-dismiss
  showCloseButton: true
});

// Promise Toast (async operations)
Swal.toast.promise(
  fetch('/api/save-data'),
  {
    loading: 'Saving...',
    success: 'Data saved successfully!',
    error: 'Failed to save data.'
  }
);

// Alternative Promise Toast
const savePromise = saveData();
Swal.toast({
  icon: 'loading',
  title: 'Saving...'
}).then(async (toast) => {
  try {
    await savePromise;
    toast.update({ icon: 'success', title: 'Saved!' });
  } catch (e) {
    toast.update({ icon: 'error', title: 'Failed!' });
  }
});

// Multiple Toasts
['File uploaded', 'Processing complete', 'Email sent'].forEach((msg, i) => {
  setTimeout(() => {
    Swal.toast({ icon: 'success', title: msg });
  }, i * 500);
});

// Dismiss All Toasts
Swal.toast.dismissAll();
